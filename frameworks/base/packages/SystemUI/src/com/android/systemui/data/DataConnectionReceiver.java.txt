package com.android.systemui.data;

import android.content.BroadcastReceiver;
import android.content.Context;
import android.content.Intent;
import android.os.SystemProperties;
import android.util.Log;


public class DataConnectionReceiver extends BroadcastReceiver {

    private static final String TAG = "DataConnectionReceiver";
    private static final Boolean DBG = true;
    //laichunjiang@wind-mobi.com 20180604 begin
    private static final String ACTION_WIFI_DATA_SERVICE_FAILOVER_DIALOG
        = "com.systemui.intent.action.WIFI_DATA_SERVICE_FAILOVER_DIALOG";
    //laichunjiang@wind-mobi.com 20180604 end
    @Override
    public void onReceive(Context context, Intent intent) {
        if (intent == null) {
            return;
        }
        String action = intent.getAction();
        if (DBG) {
            Log.d(TAG, "Receive " + action);
        }

        if (!ACTION_WIFI_DATA_SERVICE_FAILOVER_DIALOG.equals(action)) {
            return;
        }

        Intent start = new Intent(context, DataConnectionDialog.class);
        start.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        context.startActivity(start);
    }
}
