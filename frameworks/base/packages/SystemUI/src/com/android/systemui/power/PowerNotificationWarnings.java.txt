    @Override
    public void update(int batteryLevel, int bucket, long screenOffTime) {
        mBatteryLevel = batteryLevel;
        if (bucket >= 0) {
            mBucketDroppedNegativeTimeMs = 0;
        } else if (bucket < mBucket) {
            mBucketDroppedNegativeTimeMs = System.currentTimeMillis();
        }
        mBucket = bucket;
        mScreenOffTime = screenOffTime;
    }

    private void updateNotification() {
        if (DEBUG) Slog.d(TAG, "updateNotification mWarning=" + mWarning + " mPlaySound="
                + mPlaySound + " mInvalidCharger=" + mInvalidCharger);
        if (mInvalidCharger) {
            showInvalidChargerNotification();
            mShowing = SHOWING_INVALID_CHARGER;
        } else if (mWarning) {
            showWarningNotification();
            mShowing = SHOWING_WARNING;
            //shiyaru@wind-mobi.com fix Feature #10475 20180331 start
            switch (mBucket) {
                case -1:
                case -2:
                    showWarningDialog();
                    break;
                case -3:
                    startShutDownTimer();
                    showShutdownDialog();
                    break;
                default:
                    Slog.d(TAG,"nomal");
                    break;
             }
            //shiyaru@wind-mobi.com fix Feature #10475 20180331 end
        } else {
            mNoMan.cancelAsUser(TAG_BATTERY, SystemMessage.NOTE_BAD_CHARGER, UserHandle.ALL);
            mNoMan.cancelAsUser(TAG_BATTERY, SystemMessage.NOTE_POWER_LOW, UserHandle.ALL);
            //shiyaru@wind-mobi.com fix TABFIVEP-4070 2018.9.29 start
            if(mlowBatteryDialogConfirmation != null) {
                mlowBatteryDialogConfirmation.dismiss();
                mlowBatteryDialogConfirmation = null;
            }
            //shiyaru@wind-mobi.com fix TABFIVEP-4070 2018.9.29 end
            mShowing = SHOWING_NOTHING;
        }
    }

    //shiyaru@wind-mobi.com fix Feature #10475 20180331 start
    private void showWarningDialog(){
        boolean flag = mlowBatteryDialogConfirmation != null;
        Slog.d(TAG," flag="+flag);
        if (mlowBatteryDialogConfirmation != null) return;
        Slog.d(TAG," mlowBatteryDialogConfirmation=");
        boolean mSaver = mPowerMan.isPowerSaveMode();
        final int textRes = mSaver ? R.string.battery_low_percent_format_saver_started : R.string.battery_low_percent_format;
        final SystemUIDialog d = new SystemUIDialog(mContext);
        View v = View.inflate(mContext, R.layout.battery_low, null);
        mBatteryLevelTextView = (TextView)v.findViewById(R.id.level_percent);
       if(mBatteryLevel > 10){
           return;
       }
       final String percentage = NumberFormat.getPercentInstance().format((double) mBatteryLevel / 100.0);
        mBatteryLevelTextView.setText(mContext.getString(textRes, percentage));
        d.setTitle(mContext.getString(R.string.charge_need_connect));
        d.setView(v);
        if(!mSaver){
            d.setNegativeButton(R.string.battery_saver_start_action, startSaveBattery);
        }
        d.setPositiveButton(android.R.string.ok, null);
        d.setShowForAllUsers(true);
        d.setCanceledOnTouchOutside(false);
        d.show();
        mlowBatteryDialogConfirmation = d;
    }

    private final OnClickListener startSaveBattery = new OnClickListener() {
        @Override
        public void onClick(DialogInterface dialog, int which) {
                AsyncTask.execute(new Runnable() {
                        @Override
                        public void run() {
                           Intent intent= new Intent(ACTION_START_SAVER);
                           mContext.sendBroadcastAsUser(intent,UserHandle.ALL);
                        }
                });
        }
    };

    void startShutDownTimer(){
        Slog.i(TAG,"start auto shut down timer");
        if(mTimer == null) {
            mTimer = new CountDownTimer(30000,1000) {
                @Override
                public void onTick(long second ){
                    Slog.i(TAG,"CountDownTimer");
                }
                @Override
                public void onFinish(){
                    Slog.i(TAG,"time is out and shut down timer");
                    shutDown();
                }
            }.start();
        }
    }

    void shutDown(){
        Intent intent = new Intent(Intent.ACTION_REQUEST_SHUTDOWN);
        intent.putExtra(Intent.EXTRA_KEY_CONFIRM, false);
        intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        mContext.startActivity(intent);
    }

    private void showShutdownDialog(){
       Slog.i(TAG,"show auto shut down dialog");
       if (mshutDownDialogConfirmation != null) return;
       final SystemUIDialog d = new SystemUIDialog(mContext);
       d.setTitle(R.string.shut_down);
       d.setMessage(R.string.auto_shutdown_message);
       d.setIconAttribute(android.R.attr.alertDialogIcon);
       d.setPositiveButton(android.R.string.ok, null);
       d.setShowForAllUsers(true);
       d.setCanceledOnTouchOutside(false);
       d.setOnDismissListener(new OnDismissListener() {
            @Override
            public void onDismiss(DialogInterface dialog) {
                mshutDownDialogConfirmation = null;
            }
        });
        d.show();
        mshutDownDialogConfirmation = d;
    }

    private final OnClickListener shutdown = new OnClickListener() {
        @Override
        public void onClick(DialogInterface dialog, int which) {
                AsyncTask.execute(new Runnable() {
                        @Override
                        public void run() {
                              cancelTimer();
                               shutDown();;
                        }
                });
        }
    };

    void cancelTimer(){
        if( mTimer != null ){
                Slog.i(TAG,"cancel the timer");
                mTimer.cancel();
                mTimer = null;
        }
    }

    @Override
    public void dismissShutDownWarning() {
        if (mshutDownDialogConfirmation != null) {
            mshutDownDialogConfirmation.dismiss();
            mshutDownDialogConfirmation = null;
            cancelTimer();
        }
    }

    @Override
    public void dismissTempOnChargeWarning() {
        if (!mTempOnChargeWarning) {
            return;
        }
        mTempOnChargeWarning = false;
        dismissTempOnChargeWarningInternal();
    }

    private void dismissTempOnChargeWarningInternal() {
        mNoMan.cancelAsUser(TAG_TEMP_ONCHARGE, SystemMessage.NOTE_HIGH_TEMP, UserHandle.ALL);
    }

    @Override
    public void showTempOnChargeWarning(boolean isHigh) {
        if (mTempOnChargeWarning) {
            return;
        }
        int textRes;
        if (isHigh) {
            textRes = R.string.temp_high_message;
        } else {
            textRes = R.string.temp_low_message;
        }
        showTempOnChargeDialog(textRes);
        mTempOnChargeWarning = true;
        final Notification.Builder nb =
                new Notification.Builder(mContext, NotificationChannels.ALERTS)
                        .setSmallIcon(R.drawable.ic_device_thermostat_24)
                        .setWhen(0)
                        .setShowWhen(false)
                        .setContentTitle(mContext.getString(R.string.temp_oncharge_title))
                        .setContentText(mContext.getString(textRes))
                        .setOnlyAlertOnce(true)
                        .setOngoing(true)
                        .setVisibility(Notification.VISIBILITY_PUBLIC)
                        .setDeleteIntent(pendingBroadcast(ACTION_DISMISSED_TEMP_ONCHARGE_WARNING))
                        .setColor(Utils.getColorAttr(mContext, android.R.attr.colorError));
        SystemUI.overrideNotificationAppName(mContext, nb);
        final Notification n = nb.build();
        mNoMan.notifyAsUser(TAG_TEMP_ONCHARGE, SystemMessage.NOTE_HIGH_TEMP, n, UserHandle.ALL);
    }

    public void showTempOnChargeDialog(int res){
        if (mShowTempOnChargeDialog != null || mTempOnChargeWarning) return;
        final SystemUIDialog d = new SystemUIDialog(mContext);
        d.setTitle(R.string.temp_oncharge_title);
        d.setMessage(res);
        d.setPositiveButton(android.R.string.ok, dialogdissmiss);
        d.setIconAttribute(android.R.attr.alertDialogIcon);
        d.setShowForAllUsers(true);
        d.setCanceledOnTouchOutside(false);
        d.show();
        mShowTempOnChargeDialog = d;
        startTempTimer();
    }

    public void dismissTempDialog() {
        if(mShowTempOnChargeDialog != null) {
                mShowTempOnChargeDialog.dismiss();
            mShowTempOnChargeDialog = null;
        }
        if(mTempTimer != null) {
            mTempTimer.cancel();
            mTempTimer = null;
        }
    }

        if(mTempTimer == null) {
           mTempTimer = new CountDownTimer(3000,1000) {
               @Override
               public void onTick(long second ){
                      Slog.i(TAG,"CountDownTimer");
               }
               @Override
               public void onFinish(){
                   dismissTempDialog();
               }
           }.start();
        }
    }

    private final OnClickListener dialogdissmiss = new OnClickListener() {
        @Override
        public void onClick(DialogInterface dialog, int which) {
            AsyncTask.execute(new Runnable() {
                @Override
                public void run() {
                    dismissTempDialog();
                }
            });
        }
    };

    private void showWarningNotification() {
        final int textRes = R.string.battery_low_percent_format;
        final String percentage = NumberFormat.getPercentInstance().format((double) mBatteryLevel / 100.0);

        final Notification.Builder nb =
                new Notification.Builder(mContext, NotificationChannels.BATTERY)
                        .setSmallIcon(R.drawable.ic_power_low)
                        // Bump the notification when the bucket dropped.
                        .setWhen(mBucketDroppedNegativeTimeMs)
                        .setShowWhen(false)
                        .setContentTitle(mContext.getString(R.string.battery_low_title))
                        .setContentText(mContext.getString(textRes, percentage))
                        .setOnlyAlertOnce(true)
                        .setDeleteIntent(pendingBroadcast(ACTION_DISMISSED_WARNING))
                        .setVisibility(Notification.VISIBILITY_PUBLIC)
                        .setColor(Utils.getColorAttr(mContext, android.R.attr.colorError));
        if (hasBatterySettings()) {
            nb.setContentIntent(pendingBroadcast(ACTION_SHOW_BATTERY_SETTINGS));
        }
        nb.addAction(0,
                mContext.getString(R.string.battery_saver_start_action),
                pendingBroadcast(ACTION_START_SAVER));
        nb.setOnlyAlertOnce(!mPlaySound);
        mPlaySound = false;
        SystemUI.overrideNotificationAppName(mContext, nb);
        final Notification n = nb.build();
        mNoMan.cancelAsUser(TAG_BATTERY, SystemMessage.NOTE_BAD_CHARGER, UserHandle.ALL);
        mNoMan.notifyAsUser(TAG_BATTERY, SystemMessage.NOTE_POWER_LOW, n, UserHandle.ALL);
    }

    private PendingIntent pendingBroadcast(String action) {
        return PendingIntent.getBroadcastAsUser(mContext,
                0, new Intent(action), 0, UserHandle.CURRENT);
    }

    private static Intent settings(String action) {
        return new Intent(action).setFlags(Intent.FLAG_ACTIVITY_NEW_TASK
                | Intent.FLAG_ACTIVITY_MULTIPLE_TASK
                | Intent.FLAG_ACTIVITY_EXCLUDE_FROM_RECENTS
                | Intent.FLAG_ACTIVITY_NO_HISTORY
                | Intent.FLAG_ACTIVITY_CLEAR_TOP);
    }

    }

    private void dismissInvalidChargerNotification() {
        if (mInvalidCharger) Slog.i(TAG, "dismissing invalid charger notification");
        mInvalidCharger = false;
        updateNotification();
    }

    @Override
    public void showInvalidChargerWarning() {
        mInvalidCharger = true;
        updateNotification();
    }

    @Override
    public void userSwitched() {
        updateNotification();
    }

    private void showStartSaverConfirmation() {
        if (mSaverConfirmation != null) return;
        final SystemUIDialog d = new SystemUIDialog(mContext);
        d.setTitle(R.string.battery_saver_confirmation_title);
        d.setMessage(com.android.internal.R.string.battery_saver_description);
        d.setNegativeButton(android.R.string.cancel, null);
        d.setPositiveButton(R.string.battery_saver_confirmation_ok, mStartSaverMode);
        d.setShowForAllUsers(true);
        d.setOnDismissListener(new OnDismissListener() {
            @Override
            public void onDismiss(DialogInterface dialog) {
                mSaverConfirmation = null;
            }
        });
        d.show();
        mSaverConfirmation = d;
    }

    private void setSaverMode(boolean mode) {
        mPowerMan.setPowerSaveMode(mode);
    }

    private final class Receiver extends BroadcastReceiver {

        public void init() {
            IntentFilter filter = new IntentFilter();
            filter.addAction(ACTION_SHOW_BATTERY_SETTINGS);
            filter.addAction(ACTION_START_SAVER);
            filter.addAction(ACTION_DISMISSED_WARNING);
            filter.addAction(ACTION_CLICKED_TEMP_WARNING);
            filter.addAction(ACTION_DISMISSED_TEMP_WARNING);
            filter.addAction(ACTION_CLICKED_THERMAL_SHUTDOWN_WARNING);
            filter.addAction(ACTION_DISMISSED_THERMAL_SHUTDOWN_WARNING);
            mContext.registerReceiverAsUser(this, UserHandle.ALL, filter,
                    android.Manifest.permission.STATUS_BAR_SERVICE, mHandler);
        }

        @Override
        public void onReceive(Context context, Intent intent) {
            final String action = intent.getAction();
            Slog.i(TAG, "Received " + action);
            if (action.equals(ACTION_SHOW_BATTERY_SETTINGS)) {
                dismissLowBatteryNotification();
                mContext.startActivityAsUser(mOpenBatterySettings, UserHandle.CURRENT);
            } else if (action.equals(ACTION_START_SAVER)) {
                dismissLowBatteryNotification();
                showStartSaverConfirmation();
            } else if (action.equals(ACTION_DISMISSED_WARNING)) {
                dismissLowBatteryWarning();
            } else if (ACTION_CLICKED_TEMP_WARNING.equals(action)) {
                dismissHighTemperatureWarningInternal();
                showHighTemperatureDialog();
            } else if (ACTION_DISMISSED_TEMP_WARNING.equals(action)) {
                dismissHighTemperatureWarningInternal();
            } else if (ACTION_CLICKED_THERMAL_SHUTDOWN_WARNING.equals(action)) {
                dismissThermalShutdownWarning();
                showThermalShutdownDialog();
            } else if (ACTION_DISMISSED_THERMAL_SHUTDOWN_WARNING.equals(action)) {
                dismissThermalShutdownWarning();
            }
        }
    }

    private final OnClickListener mStartSaverMode = new OnClickListener() {
        @Override
        public void onClick(DialogInterface dialog, int which) {
            AsyncTask.execute(new Runnable() {
                @Override
                public void run() {
                    setSaverMode(true);
                }
            });
        }
    };
}
