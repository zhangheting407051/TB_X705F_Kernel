    private View.OnClickListener mOnCropClickListener = new View.OnClickListener() {
        @Override
        public void onClick(View v) {
            int index = (int) v.getTag();
            Log.e(TAG, "OnCropClickListener index = " + index);
            showRepresentations(CROP_REPRESENTATION);
            filterShowActivity.saveEditorCropState(index);
        }
    };

    private View.OnClickListener mOnStraightenClickListener = new View.OnClickListener() {
        @Override
        public void onClick(View v) {
            int index = (int) v.getTag();
            Log.e(TAG, "OnStraightenClickListener index = " + index);
            showRepresentations(STRAIGHTEN_REPRESENTATION);
            filterShowActivity.saveEditorStrightenState(index);
        }
    };

    private View.OnClickListener mOnMirrorClickListener = new View.OnClickListener() {
        @Override
        public void onClick(View v) {
            int index = (int) v.getTag();
            Log.e(TAG, "OnMirrorClickListener index = " + index);
            showRepresentations(MIRROR_REPRESENTATION);
            filterShowActivity.saveEditorMirrorState(index);
        }
    };

    private void showRepresentations(int index) {
        FiltersManager filtersManager = FiltersManager.getManager();
        mFiltersRepresentations = filtersManager.getTools();
        filterShowActivity = (FilterShowActivity) getActivity();
        filterShowActivity.showRepresentation(mFiltersRepresentations.get(index));
    }

        FiltersManager filtersManager = FiltersManager.getManager();

        //fix feature#10896 xiangkezhu@wind-mobi.com 20180319 begin
        final FilterShowActivity activity = (FilterShowActivity) getActivity();
        activity.showRepresentation(mFiltersRepresentations.get(CROP_REPRESENTATION));

    @Override
    protected void initCrops() {
        super.initCrops();
        int size = mCropLayouts.length;
        for (int i = 0; i < size ; i++) {
            LinearLayout layout = mCropLayouts[i];
            layout.setTag(i);
            layout.setOnClickListener(mOnCropClickListener);
        }
    }

    //fix feature#10896 xiangkezhu@wind-mobi.com 20180319 begin
    @Override
    protected void initStrightens() {
        super.initStrightens();
        int size = mStrightents.length;
        for (int i = 0; i < size ; i++) {
            final View view = mStrightents[i];
            view.setTag(i);
            view.setOnClickListener(mOnStraightenClickListener);
        }
    }

    @Override
    protected void initMirrors() {
        super.initMirrors();
        int size = mImageMirrors.length;
        for (int i = 0; i < size ; i++) {
            ImageView img = mImageMirrors[i];
            img.setTag(i);
            img.setOnClickListener(mOnMirrorClickListener);
        }
    }

    protected void switchFragement(int index){
        final FilterShowActivity activity = (FilterShowActivity) getActivity();
        if (mFiltersRepresentations == null) {
            FiltersManager filtersManager = FiltersManager.getManager();
            mFiltersRepresentations = filtersManager.getTools();
        }
        //fix feature#10896 xiangkezhu@wind-mobi.com 20180319 begin
        activity.showRepresentation(mFiltersRepresentations.get(index));
        //fix feature#10896 xiangkezhu@wind-mobi.com 20180319 end
    }
