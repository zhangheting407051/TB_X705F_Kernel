package com.android.gallery3d.util;

public class AlbumFolderInfo {
    public static String FOLDER_INFO_KEY = "AlbumFolder";
    public static String ATTRI_FOLDERNAME = "mFolderName";
    public static String ATTRI_ISHIDE = "isHide";
    public static String ATTRI_IS_USER_CREATE = "isUserCreate";
    public static String ATTRI_PATH = "mPath";
    public static String ATTRI_BUCKETID = "bucketId";

    private String mFolderName;
    private String mPath;
    private boolean isHide;
    private boolean isUserCreate;
    private int mBucketId;

    public AlbumFolderInfo(String folderName) {
        this(folderName, GalleryUtils.OWNER_CREATE_ALBUM_PATH + folderName, false, true, (GalleryUtils.OWNER_CREATE_ALBUM_PATH + folderName).toLowerCase().hashCode());
    }

    public AlbumFolderInfo (String folderName, String path, boolean hide, boolean userCreate, int bucketId) {
        this.mFolderName = folderName;
        this.mPath = path;
        this.isHide = hide;
        this.isUserCreate = userCreate;
        this.mBucketId = bucketId;
    }

    public String getmFolderName() {
        return mFolderName;
    }

    public void setmFolderName(String mFolderName) {
        this.mFolderName = mFolderName;
    }

    public String getmPath() {
        return mPath;
    }

    public void setmPath(String mPath) {
        this.mPath = mPath;
    }

    public boolean getIsHide() {
        return isHide;
    }

    public void setIsHide(boolean isHide) {
        this.isHide = isHide;
    }

    public boolean isUserCreate() {
        return isUserCreate;
    }

    public void setUserCreate(boolean userCreate) {
        isUserCreate = userCreate;
    }

    public int getBucketId() {
        return mBucketId;
    }

    public void setBucketId(int bucketId) {
        this.mBucketId = bucketId;
    }
}
